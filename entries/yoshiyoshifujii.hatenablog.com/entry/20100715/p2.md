def main():
    print my_last([]) == None
    print my_last("a") == None
    print my_last(["a"]) == ["a"]
    print my_last(["a", "b"]) == ["b"]
    print my_last(["a", "b", "c"]) == ["c"]
    print my_last(["a", "b", "c", "d"]) == ["d"]

def my_last(lista):
    if not lista or not isinstance(lista, list):
        return None
    return lista[len(lista)-1:]

if __name__ == "__main__":
    main()
