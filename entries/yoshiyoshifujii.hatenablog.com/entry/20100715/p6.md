def main():
    print inverte([]) == []
    print inverte("a") == []
    print inverte(["a"]) == ["a"]
    print inverte(["a", "b"]) == ["b", "a"]
    print inverte(["a", "b", "c"]) == ["c", "b", "a"]

def inverte(lista):
    if not lista or not isinstance(lista, list):
        return []
    lista.reverse()
    return lista

if __name__ == "__main__":
    main()
