def main():
    print comprimento([]) == 0
    print comprimento("a") == None
    print comprimento(["a"]) == 1
    print comprimento(["a", "b"]) == 2

def comprimento(lista):
    if not lista:
        return 0
    if not isinstance(lista, list):
        return None
    return len(lista)

if __name__ == "__main__":
    main()
