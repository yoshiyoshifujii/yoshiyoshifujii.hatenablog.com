def repli(lista, num):
    return reduce(lambda l,v: l+[v for i in range(num)], lista, []) if isinstance(lista, list) else None

if __name__ == "__main__":
    print repli("a", 3) == None
    print repli([], 0) == []
    print repli(["a"], 0) == []
    print repli(["a"], 1) == ["a"]
    print repli(["a"], 2) == ["a", "a"]
    print repli(["a", "b", "c"], 3) == ["a", "a", "a", "b", "b", "b", "c", "c", "c"]
