def encode(lista):
    pack_l = pack(lista)
    return [[len(v), v[0]] for v in pack_l] if pack_l else pack_l

def pack(lista):
    return reduce(appender2pack, lista, []) if isinstance(lista, list) else None

def appender2pack(l, v):
    if l:
        if v in l[-1]:
            l[-1].append(v)
        else:
            l.append([v])
    else:
        l.append([v])
    return l

if __name__ == "__main__":
    main()
